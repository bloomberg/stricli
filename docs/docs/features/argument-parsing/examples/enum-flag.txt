--level debug
--level warn
/// impl.ts
type Flags = {
  level: "info" | "warn" | "error";
};

export default async function({ level }: Flags) {
  console.log(`Set logging level to ${level}`);
}
/// spec.ts
import { buildCommand } from "@stricli/core";

export default buildCommand({
  loader: () => import("./impl"),
  parameters: {
    flags: {
      level: {
        kind: "enum",
        values: ["info", "warn", "error"],
        brief: "Logging severity level",
      },
    },
    aliases: {
      l: "level",
    },
  },
  docs: {
    brief: "Example for live playground with enum flag"
  },
});
/// !app.ts
import { buildApplication } from "@stricli/core";
import root from "./spec.ts";
export default buildApplication(root, { name: "run" });
