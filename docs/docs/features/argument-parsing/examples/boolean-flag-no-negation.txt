/// impl.ts
type Flags = {
  enable: boolean;
};

export default async function({ enable }: Flags) {
  console.log(enable ? "Feature is enabled" : "Feature is disabled");
}
/// spec.ts
import { buildCommand, numberParser } from "@stricli/core";

export default buildCommand({
  loader: () => import("./impl"),
  parameters: {
    flags: {
      enable: {
        kind: "boolean",
        brief: "Enable the experimental feature",
        default: true,
        withNegated: false,
      },
    },
  },
  docs: {
    brief: "Example for boolean flag with withNegated=false",
    customUsage: [
      { input: "", brief: "Uses default value (true)" },
      { input: "--enable", brief: "Explicitly enable feature" },
      { input: "--enable=false", brief: "Disable feature" },
    ],
  },
});
/// !app.ts
import { buildApplication } from "@stricli/core";
import root from "./spec.ts";
export default buildApplication(root, { name: "run" });
